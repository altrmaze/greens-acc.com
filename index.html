GLOBAL PARTNERS (CRM Module) create table partners ( id uuid default uuid_generate_v4() primary key, name text not null, company_name text, email text check (email ~* '^[A-Za-z0-9._%-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,4}$'), phone text, region text not null check (region in ('USA', 'Egypt', 'Turkey', 'Dubai', 'Other')), reputation_score int default 100 check (reputation_score >= 0 and reputation_score <= 100), created_at timestamp with time zone default timezone('utc'::text, now()) not null ); -- 2. GLOBAL INVENTORY (Multi-Hub Stock Control) create table inventory ( id uuid default uuid_generate_v4() primary key, item_name text not null, sku text unique not null, quantity int default 0 check (quantity >= 0), warehouse_location text not null, region text not null check (region in ('USA', 'Egypt', 'Turkey', 'Dubai', 'Other')), last_updated timestamp with time zone default timezone('utc'::text, now()) not null ); -- 3. FINANCIAL TRANSACTIONS (Multi-Currency Ledgers) create table transactions ( id uuid default uuid_generate_v4() primary key, partner_id uuid references partners(id) on delete set null, transaction_type text not null check (transaction_type in ('BUY', 'SELL', 'TRANSFER')), amount numeric(15, 2) not null check (amount >= 0.00), currency text default 'USD' check (currency in ('USD', 'EGP', 'TRY', 'AED')), exchange_rate numeric(10, 4) default 1.0000 check (exchange_rate > 0), status text default 'PENDING' check (status in ('PENDING', 'COMPLETED', 'CANCELLED')), created_at timestamp with time zone default timezone('utc'::text, now()) not null ); -- 4. SECURE VAULT (Encrypted Symmetrical Records) create table secure_financial_ledgers ( id uuid default uuid_generate_v4() primary key, partner_id uuid references partners(id) on delete cascade, region text not null check (region in ('USA', 'Egypt', 'Turkey', 'Dubai', 'Other')), encrypted_balance bytea not null, created_at timestamp with time zone default timezone('utc'::text, now()) not null ); -- 5. DEPLOYMENT AUDIT LOGS (Tracks GitHub & Render Status) create table deployment_logs ( id uuid default uuid_generate_v4() primary key, environment text default 'Production', provider text default 'Render', status text not null, commit_shasum text, logged_at timestamp with time zone default timezone('utc'::text, now()) not null ); -- ROW-LEVEL SECURITY (RLS) POLICIES alter table partners enable row level security; alter table inventory enable row level security; alter table transactions enable row level security; alter table secure_financial_ledgers enable row level security; create policy "Regional Isolation - Partners" on partners for all using (auth.jwt() ->> 'user_region' = region); create policy "Regional Isolation - Inventory" on inventory for all using (auth.jwt() ->> 'user_region' = region); create policy "Regional Isolation - Transactions" on transactions for all using (auth.jwt() ->> 'user_region' = region); create policy "Regional Isolation - Ledgers" on secure_financial_ledgers for all using (auth.jwt() ->> 'user_region' = region);
